agent-training.py Python
from go2 import WorkGraph, AgentTrainer # Your work patterns as training data graph = WorkGraph.connect("acme-corp") # Discover automatable processes opportunities = graph.find_automation( min_frequency=5, # per week min_time=30, # minutes each ) # Train agents on real work patterns trainer = AgentTrainer(graph) agent = trainer.build( task="invoice_processing", sidecar="claude-3.5" # enhances, not replaces )
