Simple to Create, Powerful to Use

my-skill.yaml
# A Go2 Skill that handles sales objections
name: sales-objection-handler
version: 2.1.0
description: Analyzes sales objections and suggests responses
author: go2-community

compatible_with:
  - claude
  - chatgpt
  - gemini

triggers:
  - event: objection_detected
    source: go2.call_transcript

context:
  uses:
    - go2.company_knowledge # Your documented processes
    - go2.past_interactions # What worked before
    - go2.competitor_intel # Auto-gathered insights

output:
  format: structured_response
  includes:
    - suggested_response
    - supporting_data
    - confidence_score
