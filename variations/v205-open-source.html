go2.config.yaml YAML 
# Go2 Agent Training Configuration
# Captures work data and trains agents automatically

version: "2.0"
project: "customer-support-agent"

data_sources:
  cowork_ai:
    enabled: true
    capture:
      - workflows          # How tasks are performed
      - tool_usage         # App & tool patterns
      - decision_points    # Where judgment calls happen
      - process_variations # How different people do the same task
    privacy:
      pii_redaction: true
      consent_required: true
      data_retention_days: 90

training:
  model_base: "go2/agent-foundation-v2"
  fine_tune:
    epochs: 3
    learning_rate: 2e-5
    batch_size: 16
  evaluation:
    holdout_pct: 0.15
    min_accuracy: 0.92

deployment:
  sidecar_mode: true   # Works alongside Claude, ChatGPT
  realtime_coaching: true
  auto_documentation: true
agent.py Python 
# Train an AI agent on real work data from Cowork.ai

from go2 import AgentTrainer, WorkDataset
from go2.coaching import RealtimeCoach

# Load captured work data
dataset = WorkDataset.from_cowork(
    team="customer-support",
    days=30,
    min_confidence=0.85
)

# See what Go2 discovered
print(f"Workflows mapped: {dataset.workflow_count}")
print(f"Automatable hrs/week: {dataset.automatable_hours}")
print(f"Process variations: {dataset.variations}")

# Train the agent
trainer = AgentTrainer(
    base_model="go2/agent-foundation-v2",
    dataset=dataset,
    sidecar=["claude", "chatgpt"]  # Enhance, don't replace
)

agent = trainer.train()
agent.evaluate()      # Validates against held-out data
agent.deploy()        # Goes live with real-time coaching

# Enable real-time coaching (89% worker satisfaction)
coach = RealtimeCoach(agent)
coach.start()
pipeline.yaml YAML 
# CI/CD pipeline for continuous agent improvement

name: "Go2 Agent Pipeline"
on:
  schedule: "0 2 * * 1"  # Weekly retraining
  data_threshold: 1000    # Or when 1K new observations

stages:
  - name: "collect"
    action: "go2/collect-work-data"
    params:
      source: "cowork-ai"
      filter: "new_since_last_run"

  - name: "validate"
    action: "go2/validate-data-quality"
    params:
      min_quality_score: 0.9
      check_privacy: true

  - name: "retrain"
    action: "go2/incremental-train"
    params:
      strategy: "incremental"
      rollback_on_regression: true

  - name: "deploy"
    action: "go2/canary-deploy"
    params:
      canary_pct: 10
      promotion_after: "24h"
